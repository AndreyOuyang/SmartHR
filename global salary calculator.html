  <script>

    // 税率数据

    const taxData = {

      USA: { employer_rate: 0.10, employee_social: 0.08, tax_single: 0.22, tax_married: 0.18 },

      France: { employer_rate: 0.45, employee_social: 0.22, tax_single: 0.20, tax_married: 0.14 },

      UK: { employer_rate: 0.13, employee_social: 0.12, tax_single: 0.20, tax_married: 0.18 },

      Germany: { employer_rate: 0.21, employee_social: 0.20, tax_single: 0.25, tax_married: 0.20 },

      Spain: { employer_rate: 0.30, employee_social: 0.18, tax_single: 0.22, tax_married: 0.18 },

      Italy: { employer_rate: 0.30, employee_social: 0.19, tax_single: 0.23, tax_married: 0.20 },

      Netherlands: { employer_rate: 0.18, employee_social: 0.15, tax_single: 0.25, tax_married: 0.20 },

      India: { employer_rate: 0.18, employee_social: 0.12, tax_single: 0.20, tax_married: 0.15 },

      Japan: { employer_rate: 0.16, employee_social: 0.14, tax_single: 0.22, tax_married: 0.18 },

      Brazil: { employer_rate: 0.28, employee_social: 0.11, tax_single: 0.22, tax_married: 0.18 }

    };


    // 国家名称映射

    const countryMap = {

      en: {

        "USA": "USA", "France": "France", "UK": "UK", "Germany": "Germany", 

        "Spain": "Spain", "Italy": "Italy", "Netherlands": "Netherlands", 

        "India": "India", "Japan": "Japan", "Brazil": "Brazil"

      },

      cn: {

        "美国 (USA)": "USA", "法国 (France)": "France", "英国 (UK)": "UK", 

        "德国 (Germany)": "Germany", "西班牙 (Spain)": "Spain", 

        "意大利 (Italy)": "Italy", "荷兰 (Netherlands)": "Netherlands", 

        "印度 (India)": "India", "日本 (Japan)": "Japan", "巴西 (Brazil)": "Brazil"

      }

    };


    // 切换语言

    function switchLang(lang) {

      document.getElementById('en-content').classList.add('hidden');

      document.getElementById('cn-content').classList.add('hidden');

      document.getElementById(`${lang}-content`).classList.remove('hidden');

      

      document.querySelectorAll('.lang-btn').forEach(btn => btn.classList.remove('active'));

      event.target.classList.add('active');

      

      document.getElementById('result').style.display = 'none';

    }


    // 核心计算函数

    function calculate(lang) {

      const countrySelect = document.getElementById(`country-${lang}`);

      const salaryInput = document.getElementById(`salary-${lang}`);

      const marriedSelect = document.getElementById(`married-${lang}`);

      const employmentSelect = document.getElementById(`employment-${lang}`);


      const salary = parseFloat(salaryInput.value);

      if (isNaN(salary) || salary <= 0) {

        const alertMsg = lang === 'en' ? 

          "Please enter a valid positive number for monthly gross salary (USD)!" : 

          "请输入有效的月税前薪资（正数，美元）！";

        alert(alertMsg);

        salaryInput.focus();

        return;

      }


      const countryDisplay = countrySelect.value;

      const countryKey = countryMap[lang][countryDisplay];

      const countryData = taxData[countryKey];


      let employerCost = 0;

      let eorFee = 0;

      let employerCostDisplay = lang === 'en' ? "N/A (Not Applicable)" : "不适用";


      if (employmentSelect.value === "horizons") {

        eorFee = 450;

        employerCost = (salary * (1 + countryData.employer_rate)) + eorFee;

        employerCostDisplay = `$${employerCost.toFixed(2)}`;

      } else if (employmentSelect.value === "gp") {

        eorFee = 400;

        employerCost = (salary * (1 + countryData.employer_rate)) + eorFee;

        employerCostDisplay = `$${employerCost.toFixed(2)}`;

      }


      const employeeSocial = salary * countryData.employee_social;

      const taxRate = marriedSelect.value === "yes" ? countryData.tax_married : countryData.tax_single;

      const tax = salary * taxRate;

      const netSalary = salary - employeeSocial - tax;


      const resultDiv = document.getElementById('result');

      resultDiv.style.display = 'block';

      

      if (lang === 'en') {

        resultDiv.innerHTML = `

          <h3>Calculation Result (${countryDisplay})</h3>

          <p><b>Employer Total Cost:</b> ${employerCostDisplay}</p>

          <p><b>EOR Monthly Fee:</b> ${eorFee > 0 ? `$${eorFee.toFixed(2)}` : "N/A"}</p>

          <p><b>Employee Social Security:</b> $${employeeSocial.toFixed(2)}</p>

          <p><b>Income Tax Deduction:</b> $${tax.toFixed(2)}</p>

          <p><b>Employee Net Monthly Salary:</b> $${netSalary.toFixed(2)}</p>

        `;

      } else {

        resultDiv.innerHTML = `

          <h3>计算结果 (${countryDisplay})</h3>

          <p><b>雇主总成本:</b> ${employerCostDisplay}</p>

          <p><b>雇佣外包月费:</b> ${eorFee > 0 ? `$${eorFee.toFixed(2)}` : "不适用"}</p>

          <p><b>员工社保扣除:</b> $${employeeSocial.toFixed(2)}</p>

          <p><b>所得税扣除:</b> $${tax.toFixed(2)}</p>

          <p><b>员工月净薪资:</b> $${netSalary.toFixed(2)}</p>

        `;

      }

    }
